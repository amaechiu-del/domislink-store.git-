⚙️ 2. scripts/generate_lessons.py

Python
import os

OUTPUT_DIR = "docs"

CURRICULUM = {
    "primary-1-3": {
        "subjects": {
            "Mathematics": ["Numbers", "Addition", "Subtraction"],
            "English": ["Alphabet", "Reading Basics", "Simple Sentences"],
            "Basic Science": ["Living Things", "Environment", "Health"]
        }
    },
    "primary-4-6": {
        "subjects": {
            "Mathematics": ["Fractions", "Decimals", "Geometry"],
            "English": ["Comprehension", "Grammar", "Essay Writing"],
            "Basic Science": ["Energy", "Matter", "Ecosystem"],
        }
    },
    "jss-1-3": {
        "subjects": {
            "Mathematics": ["Algebra", "Geometry", "Statistics"],
            "English": ["Essay", "Summary", "Comprehension"],
            "Basic Technology": ["Tools", "Machines", "Design"],
        }
    },
    "sss-1-3": {
        "subjects": {
            "Mathematics": ["Calculus Basics", "Trigonometry", "Probability"],
            "Physics": ["Motion", "Force", "Energy"],
            "Chemistry": ["Atoms", "Reactions", "Periodic Table"],
        }
    }
}


TEMPLATE = """# {title}

## Level: {level}
## Subject: {subject}
## Topic: {topic}

---

## 🎯 Learning Objectives
- Understand {topic}
- Apply concepts in real life
- Solve related problems

---

## 🧠 Lesson Content
This lesson explains {topic} in a simple structured way.

---

## ✍️ Practice
Solve 5 questions on {topic}.

---

## 📊 Assessment
Short quiz on {topic}.
"""


def save_lesson(level, subject, topic):
    path = f"{OUTPUT_DIR}/{level}/{subject.replace(' ', '_')}"
    os.makedirs(path, exist_ok=True)

    filename = f"{path}/{topic.replace(' ', '_')}.md"

    content = TEMPLATE.format(
        title=topic,
        level=level,
        subject=subject,
        topic=topic
    )

    with open(filename, "w", encoding="utf-8") as f:
        f.write(content)


def generate_all():
    for level, data in CURRICULUM.items():
        for subject, topics in data["subjects"].items():
            for topic in topics:
                save_lesson(level, subject, topic)


if __name__ == "__main__":
    generate_all()
    print("Lesson generation complete.")

🔁 3. .github/workflows/auto-publish.yml

YAML
name: Auto Publish Lessons

on:
  push:
    branches: [ "main" ]
  schedule:
    - cron: "0 2 * * 0"   # weekly Sunday run

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Generate lessons
        run: python scripts/generate_lessons.py

      - name: Commit updates
        run: |
          git config --global user.name "auto-generator"
          git config --global user.email "actions@github.com"
          git add docs/
          git commit -m "Auto-update NERDC lessons" || echo "No changes"
          git push

function loadFlashcards() {
  fetch("flashcards.json")
    .then(res => res.json())
    .then(data => {
      let i = 0;
      showCard(data[i]);

      function showCard(card) {
        document.getElementById("content").innerHTML = `
          <h3>${card.question}</h3>
          <button onclick="alert('${card.answer}')">Show Answer</button>
        `;
      }
    });
}

function loadQuiz() {
  fetch("quiz.json")
    .then(res => res.json())
    .then(data => {
      let q = data[0];
      document.getElementById("content").innerHTML = `
        <h3>${q.question}</h3>
        ${q.options.map(o => `<button>${o}</button>`).join("")}
      `;
    });
}
